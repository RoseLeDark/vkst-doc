163 constexpr bool is_aligvalid(size_t alignment) noexcept {
164 return alignment && (alignment & (alignment - 1)) == 0u;

167 // returns the offset needed to align ptr for given alignment
168 inline size_t alig_offset(void* address, size_t alignment) noexcept {
169 if(!is_aligvalid(alignment)) return 0;
170
171 auto _isaligned = reinterpret_cast<uintptr_t>(address) & (alignment - 1);
172 return _isaligned != 0 ? (alignment - _isaligned) : 0;
173 }

175 inline size_t alignment_for(const size_t size) noexcept {
