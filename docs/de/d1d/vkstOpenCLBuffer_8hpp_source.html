82 bool assign_to_gpu(cl_command_queue queue, bool write2Device = true) {
83 std::lock_guard<lock_type> guard(m_lock);
84 cl_int err = CL_SUCCESS;
85
86 if (m_refBuffer.is_locked()) {
87 return false; // Wenn der Buffer schon gesperrt ist, abbrechen
88 }
89
90 if (write2Device && !(m_fFlags & CL_MEM_READ_ONLY)) {
91
92 if (compress_type::can_compress) {
93 // Vorbereitung für komprimierte Daten
94 void* compressed_data = nullptr;
95 size_t compressed_size = 0;
96 uint16_t error = 0;
97
98 // Komprimierung des Buffers
99 error = compress_type::compress(m_refBuffer, compressed_data, compressed_size);
100
101 // Überprüfen, ob die Kompression erfolgreich war
102 if (error != 0) {
103 VKST_LOG_ERROR("Failed to compress the buffer. Compressed data is invalid. code: " + std::to_string(error));
104 return false; // Fehler beim Komprimieren
105 }
106
107 // Schreibe die komprimierten Daten in den OpenCL-Buffer
108 err = clEnqueueWriteBuffer(queue, m_clBuffer, CL_TRUE, 0,
109 compressed_size, compressed_data,
110 0, nullptr, nullptr);
111 } else {
112 // Keine Kompression, normale Übertragung
113 err = clEnqueueWriteBuffer(queue, m_clBuffer, CL_TRUE, 0,
114 m_refBuffer.get_size_bytes(), m_refBuffer.get_raw(),
115 0, nullptr, nullptr);
116 }
117 }
118
119 if (err != CL_SUCCESS) {
120 VKST_LOG_ERROR("Failed to enqueue write buffer. Error code: " + std::to_string(err));
121 return false;
122 }
123
124 m_refBuffer.is_locked(true); // Buffer sperren
125 return true;
126 }

134 bool require(cl_command_queue queue, bool readDevice) {
135 std::lock_guard<lock_type> guard(m_lock);
136 if (!m_refBuffer.is_locked()) {
137 return false; // Wenn der Buffer nicht gesperrt ist, keine Operation
138 }
139
140
141
142 cl_int err = 0;
143
144 if (readDevice && !(m_fFlags & (CL_MEM_WRITE_ONLY)) ) {
145
146 if (compress_type::can_compress) {
147 void* compressed_mem = nullptr;
148 uint compressed_size;
149
150 // Holen der komprimierten Größe
151 err = clEnqueueReadBuffer(queue, m_clBuffer, CL_TRUE, 0,
152 sizeof(uint), &compressed_size,
153 0, nullptr, nullptr);
154 if (err != CL_SUCCESS) {
155 VKST_LOG_ERROR("Failed to read compressed size. Error code: " + std::to_string(err));
156 return false;
157 }
158 // Allokieren von Speicher für komprimierte Daten
159 compressed_mem = malloc(compressed_size);
160 if (compressed_mem == nullptr) {
161 VKST_LOG_ERROR("Memory allocation failed for compressed data.");
162 return false;
163 }
164
165 err = clEnqueueReadBuffer(queue, m_clBuffer, CL_TRUE, 0,
166 compressed_size, compressed_mem,
167 0, nullptr, nullptr);
168
169 if (err != CL_SUCCESS) {
170 free(compressed_mem); // Sicherstellen, dass der Speicher im Fehlerfall freigegeben wird
171 VKST_LOG_ERROR("Failed to read compressed data. Error code: " + std::to_string(err));
172 return false;
173 }
174
175 uint16_t error = compress_type::decompress(compressed_mem, compressed_size, m_refBuffer);
176 free(compressed_mem);
177
178 if(error != 0) {
179 VKST_LOG_ERROR("Failed to decompress the buffer. Decompressed data is invalid. code: " + std::to_string(error));
180 return false; // Fehler beim dekomprimieren
181 }
182 } else {
183
184 void* mem_temp = malloc(m_refBuffer.get_size_bytes());
185 size_t mem_size = m_refBuffer.get_size_bytes();
186
187 err = clEnqueueReadBuffer(queue, m_clBuffer, CL_TRUE, 0,
188 mem_size, mem_temp,
189 0, nullptr, nullptr);
190
191 m_refBuffer.is_locked(false); // Buffer entsperren
192
193 m_refBuffer.assign(mem_temp, mem_size);
194 free(mem_temp);
195 }
196 }
197
198 if (err != CL_SUCCESS) {
199 VKST_LOG_ERROR("Failed to enqueue read buffer. Error code: " + std::to_string(err));
200 return false;
201 }
202
203
204
205 return true;
206 }

265 void create_internal_gpu_buffer() {
266 cl_mem_flags flags = 0; // Flags, die über den Template-Parameter übergeben wurden
267 flags |= static_cast<cl_mem_flags>(m_fFlags); // Flags setzen
268
269 // Überprüfen, ob einer der Host-Flags gesetzt ist
270 m_bHostPtr = (flags & (CL_MEM_HOST_READ_ONLY | CL_MEM_HOST_WRITE_ONLY | CL_MEM_USE_HOST_PTR)) != 0;
271
272 cl_int err = 0;
273
274 // Wenn ein Host-Puffer genutzt wird, übergebe den raw-Puffer
275 m_clBuffer = clCreateBuffer(m_cContext, flags, m_refBuffer.get_size_bytes(),
276 (m_bHostPtr ? m_refBuffer.get_raw() : nullptr), &err);
277
278 if (err != CL_SUCCESS) {
279 VKST_LOG_ERROR("Failed to create OpenCL buffer. Error code: " + std::to_string(err));
280 } else {
281 VKST_LOG_INFO("OpenCL buffer created successfully.");
282 }
283 }
