214 bool resize(size_type newSize, bool bReserve = true) {
215 if(m_is2Device) { VKST_LOG_ERROR("buffer is in device"); return false; }
216
217 if(!m_bOwnMem) return false;
218 if(newSize > m_sSize) {
219 pointer __pNewRawBuffer = m_allocator.alloc(newSize * sizeof(value_type), TALIGNMENT);
220 if(__pNewRawBuffer == nullptr) {
221 VKST_LOG_ERROR("Failed to allocate memory during resize");
222 return false;
223 }
224
225 if(bReserve) memcpy(__pNewRawBuffer, m_pRawBuffer, m_sUsed * sizeof(value_type));
226
227 m_allocator.free(m_pRawBuffer);
228 m_pRawBuffer = __pNewRawBuffer;
229 m_sSize = newSize;
230
231 VKST_LOG_INFO("Buffer resized to new size: " + std::to_string(newSize));
232 }
233 m_sUsed = newSize;
234
235 return true;
236 }

245 bool change_size(size_type newSize, bool bReserve = true) {
246 if(m_is2Device) { VKST_LOG_ERROR("buffer is in device"); return false; }
247 if(!m_bOwnMem) return false;
248 if (newSize == m_sSize) return true;
249
250 pointer __pNewRawBuffer = NULL;

252 __pNewRawBuffer = m_allocator.alloc(newSize * sizeof(value_type));
253 if(bReserve) memcpy(__pNewRawBuffer, m_pRawBuffer,
254 (m_sUsed < newSize ? m_sUsed : newSize) * sizeof(value_type));
255 }
256
257 m_allocator.free(m_pRawBuffer);
258 m_pRawBuffer = __pNewRawBuffer;
259 m_sSize = newSize;
