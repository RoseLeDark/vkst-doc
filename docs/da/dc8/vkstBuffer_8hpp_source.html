139 bool resize(size_type newSize, bool bReserve = true) {
140 if(m_is2Device) { VKST_LOG_ERROR("buffer is in device"); return false; }
141
142 if(!m_bOwnMem) return false;
143 if(newSize > m_sSize) {
144 pointer __pNewRawBuffer = m_allocator.alloc(newSize * sizeof(value_type), TALIGNMENT);
145 if(__pNewRawBuffer == nullptr) {
146 VKST_LOG_ERROR("Failed to allocate memory during resize");
147 return false;
148 }
149
150 if(bReserve) memcpy(__pNewRawBuffer, m_pRawBuffer, m_sUsed * sizeof(value_type));
151
152 m_allocator.free(m_pRawBuffer);
153 m_pRawBuffer = __pNewRawBuffer;
154 m_sSize = newSize;
155
156 VKST_LOG_INFO("Buffer resized to new size: " + std::to_string(newSize));
157 }
158 m_sUsed = newSize;
159
160 return true;
161 }

170 bool change_size(size_type newSize, bool bReserve = true) {

172 if(!m_bOwnMem) return false;
173 if (newSize == m_sSize) return true;
174
175 pointer __pNewRawBuffer = NULL;
176 if(newSize > 0) {
177 __pNewRawBuffer = m_allocator.alloc(newSize * sizeof(value_type));

179 (m_sUsed < newSize ? m_sUsed : newSize) * sizeof(value_type));
